import { serve } from "https://deno.land/std@0.224.0/http/server.ts"; import { createClient } from "jsr:@supabase/supabase-js@2"; // Supabase-Konfiguration const supabaseUrl = Deno.env.get("SUPABASE_URL") || ""; const supabaseServiceKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") || ""; const supabase = createClient(supabaseUrl, supabaseServiceKey); // Google Maps API-Konfiguration const GOOGLE_API_KEY = Deno.env.get("GOOGLE_API_KEY") || ""; const GOOGLE_DIRECTIONS_URL = "https://maps.googleapis.com/maps/api/directions/json"; // Fest definierte Preiskonstanten const FIXED_PRICE_PER_KM = 0.75; // 0,75€ pro Kilometer const BASE_FEE = 3.80; // Grundgebühr von 3,80€ export default async function handler(req: Request): Promise { if (req.method !== "POST") { return new Response( JSON.stringify({ error: "Method not allowed" }), { status: 405, headers: { "Content-Type": "application/json" }, } ); } let requestBody; try { requestBody = await req.json(); } catch (error) { return new Response( JSON.stringify({ error: "Invalid JSON" }), { status: 400, headers: { "Content-Type": "application/json" }, } ); } // Erwartete Parameter const { origin, destination, calcOnly, userID, paymentId, destinationAddress, startAddress, tripStatus } = requestBody; console.log('origin: ' + JSON.stringify(origin)); console.log('destination: ' + JSON.stringify(destination)); if (origin === undefined || destination === undefined || calcOnly === undefined) { return new Response( JSON.stringify({ error: "Origin, destination und calcOnly sind erforderlich" }), { status: 400, headers: { "Content-Type": "application/json" }, } ); } // Explizite Umwandlung in Boolean, falls calcOnly als String übergeben wird const isCalcOnly = calcOnly === true || calcOnly === 'true'; // Wenn calcOnly false ist, müssen userID und paymentId vorhanden sein. if (!isCalcOnly && (!userID || !paymentId)) { return new Response( JSON.stringify({ error: "userID und paymentId sind erforderlich, wenn calcOnly false ist" }), { status: 400, headers: { "Content-Type": "application/json" }, } ); } try { console.log(`Requesting directions from ${origin} to ${destination}`); const response = await fetch( `${GOOGLE_DIRECTIONS_URL}?origin=${origin}&destination=${destination}&key=${GOOGLE_API_KEY}` ); const directionsData = await response.json(); console.log('Google Maps API response:', directionsData); const route = directionsData.routes[0]; if (!route) { return new Response( JSON.stringify({ error: "No route found" }), { status: 404, headers: { "Content-Type": "application/json" }, } ); } // Berechne die Gesamtstrecke in Kilometern const totalDistanceMeters = route.legs.reduce((sum, leg) => sum + leg.distance.value, 0); const totalDistanceKm = totalDistanceMeters / 1000; // Preisberechnung: Strecke * 0,75€ pro km + 3,80€ Grundgebühr const distanceCost = totalDistanceKm * FIXED_PRICE_PER_KM; const totalPrice = BASE_FEE + distanceCost; // Fahreranteil: 93% des Gesamtbetrags const driverEarnings = totalPrice * 0.93; console.log('totalPrice: ' + totalPrice); // PaymentID generieren: const payment_id = crypto.randomUUID(); // Wenn nur berechnet werden soll, Rückgabe der Werte if (isCalcOnly) { return new Response( JSON.stringify({ distance: totalDistanceKm, price: totalPrice, driverEarnings: driverEarnings }), { status: 200, headers: { "Content-Type": "application/json" }, } ); } else { const { data: paymentMethods, error: pmError } = await supabase .from('payment_methods') .select('paymentmeth_id') .eq('stripe_payment_method_id', paymentId); if (pmError) { throw new Error(pmError.message); } if (!paymentMethods || paymentMethods.length === 0) { throw new Error("Kein Eintrag in payment_methods für die angegebene stripe_payment_method_id gefunden."); } const selectedPaymentMethod = paymentMethods[0]; // Eintrag in die Tabelle "payments" const result = await supabase .from("payments") .insert([{ id: payment_id, customer_id: userID, fare: totalPrice, driver_earning: driverEarnings, customer_paymentmethod_id: selectedPaymentMethod.paymentmeth_id, status: "booking_process" }]); if (result.error) { console.error("DB Error:", result.error); return new Response( JSON.stringify({ error: result.error }), { status: 500, headers: { "Content-Type": "application/json" }, } ); } // Eintrag in die Tabelle "trips" const { data, error } = await supabase .from('trips') .insert([{ destination_address: destinationAddress, start_address: startAddress, destination_coordinate: destination, start_coordinate: origin, trip_distance: totalDistanceKm, trip_status: tripStatus, user_id_customer: userID, payment_id: payment_id }]); if (error) { throw error; } // Erfolgreiche Buchung: Keine Rückgabe von Werten return new Response(null, { status: 200, headers: { "Content-Type": "application/json" }, }); } } catch (error) { console.error("Error calculating/inserting route:", error); return new Response( JSON.stringify({ error: error.message }), { status: 500, headers: { "Content-Type": "application/json" }, } ); } } serve(async (req) => { if (req.method === "OPTIONS") { return new Response(null, { headers: { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "POST", "Access-Control-Allow-Headers": "Content-Type", }, }); } return handler(req); });
